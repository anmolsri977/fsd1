const box = document.getElementById("box"); const drop = document.getElementById("drop"); box.addEventListener("dragstart", function(e) { e.dataTransfer.setData("text/plain", e.target.id); }); drop.addEventListener("dragover", function(e) { e.preventDefault(); }); drop.addEventListener("drop", function(e) { e.preventDefault(); const data = e.dataTransfer.getData("text/plain"); e.target.appendChild(document.getElementById(data)); });
